Instalacion/LDAP

LDAP

Instalación

Debian y derivados

#Instalacion del servidor y cliente ldap
apt-get install slapd ldap-utils migrationtools

#Forzando un wizard de configuración
dpkg-reconfigure --priority=low slapd

#Instalando un administrador Web
apt-get install phpldapadmin

#Reiniciando el servicio
/etc/init.d/slapd restart

#Soporte para PHP
apt-get install php5-ldap

Windows

  •  Instalador
  •  Manual 
#Descomentar lo siguiente en el php.ini
extension=php_gettext.dll
extension=php_ldap.dll

#Reiniciar apache
#Iniciar el servicio
net start OpenLDAP-slapd

Configuración

#Prefijo para slapd.conf
suffix          "dc=siu,dc=edu,dc=ar"

#Login de admin
cn=admin,dc=siu,dc=edu,dc=ar

#Migración de usuarios
#Editar /etc/migrationtools/migrate_common.ph
$DEFAULT_MAIL_DOMAIN = "siu.edu.ar";
$DEFAULT_BASE = "dc=siu,dc=edu,dc=ar";  

#Ver todo lo existente
ldapsearch -x -b 'dc=siu,dc=edu,dc=ar' 'objectclass=*'
