Administrador

Nucleo

  • Algún esquema concreto para separar al menos los metadatos de usuarios, todos los sistemas en algún momento necesitan de ellos. Cosas relacionadas: Modulos, #47, #60.
  • Zonas. Forma concreta de mejorar esto: 
    <?php
    function __construct($id)
    {
        parent::__construct($id);
        $zona = toba::get_solicitud()->zona();
        if (isset($zona) && $editable = $zona->obtener_editable_propagado()){
            $zona->cargar_editable(); 
            $this->dato_que_maneja_el_ci = $editable;
        }    
    }
    
    
    function generar_interface_grafica()
    {
        $zona = toba::get_solicitud()->zona();
        if (isset($zona) && isset($this->dato_que_maneja_el_ci)) {
            $zona->obtener_html_barra_superior();
        }
        parent::generar_interface_grafica();
        if (isset($zona) && isset($this->dato_que_maneja_el_ci)) {
            $zona->obtener_html_barra_inferior();
        }    
    }        
?>
